import org.junit.Test;
import static org.junit.Assert.*;

public class HumanEval_3Test {

    @Test
    public void testBelowZero_EmptyArray() {
        assertFalse(HumanEval_3.belowZero(new int[0]));
    }

    @Test
    public void testBelowZero_UniformlyPositive() {
        assertTrue(HumanEval_3.belowZero(new int[] {1, 2, -3, 1, 2, -3}));
    }

    @Test
    public void testBelowZero_NoNegativeNumbers() {
        assertFalse(HumanEval_3.belowZero(new int[] {1, 2, -4, 5, 6}));
    }

    @Test
    public void testBelowZero_NoAndSomeNegativeNumbers() {
        assertFalse(HumanEval_3.belowZero(new int[] {1, -1, 2, -2, 5, -5, 4, -4}));
    }

    @Test
    public void testBelowZero_MostlyPositive() {
        assertTrue(HumanEval_3.belowZero(new int[] {1, -1, 2, -2, 5, -5, 4, -4}));
    }

    @Test
    public void testBelowZero_NoSingleNegative() {
        assertTrue(HumanEval_3.belowZero(new int[] {1, -2, 2, -2, 5, -5, 4, -4}));
    }
}
