from typing import List
def sumArrayGreater(data: List[float], UB: float) -> bool:
    sum = 0
    for i in range(len(data)):
        sum += data[i]
    
    if sum > UB:
        return True
    return False

public class SumArrayGreater {
    public static boolean sumArrayGreater(double[] data, double UB) {
        double sum = 0;
        for (double num : data) {
            sum += num;
        }
        return sum > UB;
    }
}
